#include <Keypad.h>
#include <SPI.h>
#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>

// ---------- TFT PIN CONFIGURATION ----------
#define TFT_CS   5
#define TFT_DC   16
#define TFT_RST  17
#define TFT_BL   4   // Backlight (can also connect directly to 3.3V)

Adafruit_ST7735 tft = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_RST);

// ---------- Keypad configuration ----------
const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};

// ESP32 GPIOs connected to keypad
byte rowPins[ROWS] = {33, 25, 26, 27}; // R1, R2, R3, R4
byte colPins[COLS] = {14, 12, 13, 32}; // C1, C2, C3, C4

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// ---------- Variables ----------
String input = "";          // Store input sequence
const int MAX_INPUT_LEN = 16; // Max characters to display

void setup() {
  Serial.begin(115200);
  Serial.println("TFT + Keypad Integration Started");

  // TFT setup
  tft.initR(INITR_BLACKTAB); // Or INITR_GREENTAB depending on module
  tft.setRotation(1);
  tft.fillScreen(ST77XX_BLACK);

  // Turn ON backlight
  pinMode(TFT_BL, OUTPUT);
  digitalWrite(TFT_BL, HIGH);

  // Display title
  tft.setTextSize(2);
  tft.setTextColor(ST77XX_CYAN);
  tft.setCursor(0, 0);
  tft.println("Keypad Input:");
  tft.drawFastHLine(0, 28, tft.width(), ST77XX_WHITE);

  // Set text for keypad input
  tft.setTextSize(3);
  tft.setTextColor(ST77XX_YELLOW);
  tft.setCursor(0, 40);
}

void loop() {
  char key = keypad.getKey();

  if (key) {
    Serial.print("Key Pressed: ");
    Serial.println(key);

    // Add key to input buffer
    if (input.length() < MAX_INPUT_LEN) {
      input += key;
    } else {
      input.remove(0, 1); // Remove oldest char if full
      input += key;
    }

    // Clear input display area
    tft.fillRect(0, 40, tft.width(), 60, ST77XX_BLACK);

    // Show updated input
    tft.setCursor(0, 40);
    tft.print(input);
  }
}
