/*
  GeoLinker-ready GPS Tracker
  - Posts JSON to https://www.circuitdigest.cloud/geolinker
  - Offline buffer using SPIFFS
  - NTP time + timezone offset
  - TinyGPS++ for NMEA parsing
*/

#include <WiFi.h>
#include <HTTPClient.h>
#include <TinyGPSPlus.h>
#include <SPIFFS.h>
#include <time.h>
#include <WiFiClientSecure.h>

//
// ========== CONFIG ==========
//
const char* WIFI_SSID = "12345678";
const char* WIFI_PASS = "12345678";

const char* GEO_BASE_HTTPS = "https://www.circuitdigest.cloud/geolinker";
const char* GEO_BASE_HTTP  = "http://www.circuitdigest.cloud/geolinker"; // fallback if HTTPS fails
const char* API_KEY = "KKs6rNRt4x16";            // your GeoLinker API key
const char* DEVICE_ID = "testdevice";            // your device id (must match UI)

const int GPS_RX = 34; // GPS TX -> ESP32 RX
const int GPS_TX = 35; // GPS RX -> ESP32 TX
const uint32_t GPS_BAUD = 9600;

const uint32_t SEND_INTERVAL_MS = 15000;    // send every 5 seconds when GPS updates (tune as needed)
const size_t   OFFLINE_BUFFER_LIMIT = 100; // max stored records
const char* OFFLINE_FILE = "/offline.txt"; // SPIFFS newline-separated JSON entries

// timezone offset (hours, minutes) - example IST = +5:30
const int8_t TZ_HOURS = 5;
const int8_t TZ_MINUTES = 30;

//
// ========== GLOBALS ==========
//
TinyGPSPlus gps;
HardwareSerial gpsSerial(2);

unsigned long lastSendMs = 0;
bool wifiConnected = false;

//
// ========== UTIL: Time / Timestamp ==========
bool initNTP()
{
  // gmtOffset = seconds east of UTC (i.e. UTC+5:30 => +19800)
  long gmtOffset_sec = (long)TZ_HOURS * 3600 + (long)TZ_MINUTES * 60;
  const long daylightOffset_sec = 0;
  configTime(gmtOffset_sec, daylightOffset_sec, "pool.ntp.org", "time.google.com");
  // wait a little for sync (non-blocking)
  for (int i=0;i<8;i++){
    struct tm timeinfo;
    if (getLocalTime(&timeinfo)) return true;
    delay(500);
  }
  return false;
}

String getTimestampString() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo)) {
    // fallback: epoch
    return String("1970-01-01 00:00:00");
  }
  char buf[32];
  // YYYY-MM-DD HH:MM:SS
  strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", &timeinfo);
  return String(buf);
}

//
// ========== UTIL: SPIFFS offline buffer ==========
bool initSPIFFS() {
  if (!SPIFFS.begin(true)) {
    Serial.println("SPIFFS mount failed");
    return false;
  }
  return true;
}

bool appendOffline(const String &jsonLine) {
  // ensure we don't blow memory: if file too large, trim oldest entries
  File f = SPIFFS.open(OFFLINE_FILE, FILE_APPEND);
  if (!f) {
    Serial.println("Failed open offline file for append");
    return false;
  }
  f.println(jsonLine);
  f.close();
  Serial.println("Buffered offline payload");
  return true;
}

bool flushOfflineToServer(); // forward declaration

//
// ========== UTIL: Send JSON to GeoLinker ==========
bool postToGeoLinkerHTTPS(const String &payload, String &respBody, int &respCode) {
  WiFiClientSecure client;
  client.setInsecure(); // skip certificate validation (practical for hobby use)
  HTTPClient http;
  http.setTimeout(15000);
  if (!http.begin(client, GEO_BASE_HTTPS)) {
    Serial.println("[HTTP] begin failed (HTTPS)");
    return false;
  }
  http.addHeader("Content-Type", "application/json");
  http.addHeader("Authorization", API_KEY);
  respCode = http.POST(payload);
  respBody = (respCode>0) ? http.getString() : String();
  http.end();
  return (respCode > 0);
}

bool postToGeoLinkerHTTP(const String &payload, String &respBody, int &respCode) {
  WiFiClient client;
  HTTPClient http;
  http.setTimeout(15000);
  if (!http.begin(client, GEO_BASE_HTTP)) {
    Serial.println("[HTTP] begin failed (HTTP)");
    return false;
  }
  http.addHeader("Content-Type", "application/json");
  http.addHeader("Authorization", API_KEY);
  respCode = http.POST(payload);
  respBody = (respCode>0) ? http.getString() : String();
  http.end();
  return (respCode > 0);
}

bool sendPayloadToServer(const String &payload) {
  // Try HTTPS first (insecure TLS)
  String resp;
  int code = -999;
  Serial.println("Posting to GeoLinker (HTTPS)...");
  if (postToGeoLinkerHTTPS(payload, resp, code)) {
    Serial.printf("HTTPS respCode=%d\n", code);
    Serial.println("Body: " + resp);
    if (code==200 || code==201) return true;
    // if server returned an error (e.g., 400/401), don't retry endlessly client-side
    if (code >= 400 && code < 500) {
      Serial.println("Server rejected payload (client error).");
      return false;
    }
  } else {
    Serial.println("HTTPS attempt failed, will try HTTP fallback.");
  }

  // Fallback to HTTP
  Serial.println("Posting to GeoLinker (HTTP)...");
  resp = "";
  code = -999;
  if (postToGeoLinkerHTTP(payload, resp, code)) {
    Serial.printf("HTTP respCode=%d\n", code);
    Serial.println("Body: " + resp);
    if (code==200 || code==201) return true;
    if (code >= 400 && code < 500) return false;
  } else {
    Serial.println("HTTP attempt failed (network/dns).");
  }

  return false;
}

//
// ========== BUILD payload & send (and buffer if fail) ==========
void queueOrSend(double lat, double lon) {
  String timestamp = getTimestampString();
  // Build JSON according to GeoLinker doc: arrays
  String payload = String("{\"device_id\":\"") + DEVICE_ID + "\",\"timestamp\":[\"" + timestamp + "\"],\"lat\":[" + String(lat, 6) + "],\"long\":[" + String(lon, 6) + "]}";

  Serial.println("Prepared payload: ");
  Serial.println(payload);

  if (WiFi.status() == WL_CONNECTED) {
    bool ok = sendPayloadToServer(payload);
    if (ok) {
      Serial.println("✓ payload accepted by GeoLinker");
      // try flush offline
      flushOfflineToServer();
      return;
    } else {
      Serial.println("✗ payload NOT accepted; buffering offline");
      appendOffline(payload);
    }
  } else {
    Serial.println("WiFi not connected -> buffering");
    appendOffline(payload);
  }
}

bool flushOfflineToServer() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("Cannot flush offline: WiFi disconnected");
    return false;
  }
  if (!SPIFFS.exists(OFFLINE_FILE)) {
    // nothing to flush
    return true;
  }
  File f = SPIFFS.open(OFFLINE_FILE, FILE_READ);
  if (!f) {
    Serial.println("Failed to open offline file for reading");
    return false;
  }
  // Read all lines and attempt to send one by one. We'll rebuild file of remaining lines.
  std::vector<String> leftover;
  while (f.available()) {
    String line = f.readStringUntil('\n');
    line.trim();
    if (line.length()==0) continue;
    Serial.println("Attempting flush payload: " + line);
    bool ok = sendPayloadToServer(line);
    if (!ok) {
      leftover.push_back(line); // keep for later
      // If first attempt fails, we still try remaining lines - but if network flaky it'll fill again.
    } else {
      Serial.println("Flushed offline payload");
    }
  }
  f.close();

  // rewrite leftover to file
  if (leftover.size() == 0) {
    SPIFFS.remove(OFFLINE_FILE);
    Serial.println("Offline buffer emptied.");
    return true;
  } else {
    File fw = SPIFFS.open(OFFLINE_FILE, FILE_WRITE);
    if (!fw) {
      Serial.println("Failed to rewrite offline file");
      return false;
    }
    for (auto &s : leftover) {
      fw.println(s);
    }
    fw.close();
    Serial.printf("Offline buffer left: %u entries\n", (unsigned)leftover.size());
    return false;
  }
}

//
// ========== SETUP & LOOP ==========
void setup() {
  Serial.begin(115200);
  delay(200);
  Serial.println("\n=== GeoLinker GPS Tracker Starting ===");

  // init SPIFFS
  if (!initSPIFFS()) {
    Serial.println("Warning: SPIFFS init failed (offline buffering disabled)");
  }

  // Start GPS serial
  gpsSerial.begin(GPS_BAUD, SERIAL_8N1, GPS_RX, GPS_TX);
  Serial.printf("GPS serial started on RX=%d TX=%d @%u\n", GPS_RX, GPS_TX, GPS_BAUD);

  // Connect WiFi
  Serial.printf("Connecting to WiFi '%s' ...\n", WIFI_SSID);
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  unsigned long start = millis();
  while (millis() - start < 15000) {
    if (WiFi.status() == WL_CONNECTED) break;
    Serial.print(".");
    delay(500);
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi connected. IP: " + WiFi.localIP().toString());
    wifiConnected = true;
  } else {
    Serial.println("\nWiFi not connected (will buffer offline).");
    wifiConnected = false;
  }

  // init NTP (to generate timestamps): try even if wifiConnected false (it will fail)
  if (wifiConnected) {
    if (initNTP()) Serial.println("NTP time synced.");
    else Serial.println("NTP sync failed (will retry later).");
  } else {
    Serial.println("NTP skipped (no WiFi).");
  }

  // Try flushing any existing offline data if WiFi available
  if (wifiConnected) flushOfflineToServer();

  Serial.println("Setup complete. Waiting for GPS data...");
}

void loop() {
  // Read NMEA bytes and feed TinyGPS
  while (gpsSerial.available() > 0) {
    char c = gpsSerial.read();
    gps.encode(c);
    // echo raw NMEA optionally:
    // Serial.write(c);
  }

  // If we have a newly updated location and interval elapsed, queue/send
  if (gps.location.isUpdated()) {
    unsigned long now = millis();
    if (now - lastSendMs >= SEND_INTERVAL_MS) {
      lastSendMs = now;
      if (gps.location.isValid()) {
        double lat = gps.location.lat();
        double lon = gps.location.lng();
        Serial.printf("GPS: %f, %f\n", lat, lon);
        queueOrSend(lat, lon);
      } else {
        Serial.println("GPS location not valid yet.");
      }
    }
  }

  // keep WiFi alive: attempt reconnect if disabled and auto-reconnect is desired
  if (WiFi.status() != WL_CONNECTED) {
    static unsigned long lastConnectAttempt = 0;
    if (millis() - lastConnectAttempt > 15000) {
      lastConnectAttempt = millis();
      Serial.println("Attempting WiFi reconnect...");
      WiFi.disconnect();
      WiFi.begin(WIFI_SSID, WIFI_PASS);
      unsigned long tstart = millis();
      while (millis() - tstart < 8000) {
        if (WiFi.status() == WL_CONNECTED) break;
        delay(200);
      }
      if (WiFi.status() == WL_CONNECTED) {
        Serial.println("WiFi reconnected.");
        initNTP();
        flushOfflineToServer();
      } else {
        Serial.println("WiFi reconnect failed.");
      }
    }
  }

  delay(100);
}
